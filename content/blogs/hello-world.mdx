Here's a simple Python snippet:

```python
def fibonacci(n):
    a, b = 0, 1
    for _ in range(n):
        a, b = b, a + b
    return a

```

  let timer;
  return (...args) => {

    timer = setTimeout(() => fn(...args), ms);
  };
};
